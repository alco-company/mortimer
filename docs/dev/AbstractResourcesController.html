def remove_image
  unless authorize(:destroy)
    not_authorized
  else
    if (resource.respond_to?( :images ) && !params[:image_id].blank?)
      resource.images.find(params[:image_id]).purge 
    end
    respond_to do |format|
      format.json { head 200 }
    end
  end
end
